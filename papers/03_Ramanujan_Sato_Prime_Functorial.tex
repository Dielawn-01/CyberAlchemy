def multiplicative_order(a: int, p: int) -> int:
    """Computes the exact multiplicative order of a in F_p*."""
    if math.gcd(a, p) != 1:
        return 0
    order = 1
    val = a % p
    while val != 1:
        val = (val * a) % p
        order += 1
        if order > p:
            raise ValueError(f"Order overflow for {a} mod {p}")
    return order

    return roots

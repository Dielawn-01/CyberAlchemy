\subsection{The Tryptophan Fork (Transport-Grounded)}

\begin{definition}[Tryptophan pool]
A tryptophan pool $W = (\tau, s, k)$ consists of total tryptophan
$\tau > 0$ (mol\,m$^{-3}$), serotonin fraction $s \in [0,1]$, and
kynurenine fraction $k \in [0,1]$, subject to $s + k = 1$.
\end{definition}

\begin{definition}[Mycobiome diversion]
\label{def:diversion}
The diversion parameter $\delta \in [0,1]$ represents the fraction of
serotonin-bound tryptophan redirected to kynurenine by fungal IDO/TDO
activity. In transport terms, $\delta$ maps to the \emph{Michaelis--Menten}
rate $v = k_\mathrm{cat}[\mathrm{E}][\mathrm{S}]/(K_m + [\mathrm{S}])$.
\end{definition}

\begin{theorem}[Fork mass balance (PNP-grounded)]
\label{thm:mass}
For any tryptophan pool $W$:
$\mathrm{serotonin\_substrate}(W) + \mathrm{kynurenine\_substrate}(W) = W.\tau$.

\begin{proof}
\emph{Derivation A (algebraic).} By expansion: $\tau s + \tau k = \tau(s+k) = \tau$.

\emph{Derivation B (transport, RJ2 Eq.~1).}
The continuity equation $\partial_t c + \nabla\cdot\mathbf{J} = R$ at
steady state gives $\nabla\cdot\mathbf{J} = R$. Integrating over the
gut lumen volume $V$: total flux out = total reaction. Since tryptophan
is neither created nor destroyed at the fork, the outgoing serotonin
flux plus outgoing kynurenine flux equals the incoming tryptophan flux.

\emph{Dimensional check.}
$[\partial_t c] = \mathrm{mol\,m^{-3}\,s^{-1}} = [\nabla\cdot\mathbf{J}] = [R]$. \checkmark

\emph{SI benchmark.}
At $\tau = 40\;\mu\mathrm{mol/L}$ (mid-range plasma Trp~\cite{fernstrom71}),
$\delta = 0.3$: serotonin substrate $= 28\;\mu\mathrm{mol/L}$,
kynurenine substrate $= 12\;\mu\mathrm{mol/L}$.

Verified: \texttt{verify\_mmm\_v2.py} \S A (101-point sweep, mass balance
at every point).
\qedhere
\end{proof}
\end{theorem}

\begin{theorem}[Diversion reduces serotonin (IDO-grounded)]
\label{thm:diversion}

$\mathrm{serotonin\_substrate}(W_2) < \mathrm{serotonin\_substrate}(W_1)$.

\begin{proof}
\emph{Derivation A (algebraic).} From $s + k = 1$ and $\tau > 0$.

\emph{Dimensional check.}

$[v] = [k_\mathrm{cat}][\mathrm{E}][\mathrm{S}]/[K_m] = \mathrm{s^{-1}\cdot mol\,m^{-3}} = \mathrm{mol\,m^{-3}\,s^{-1}}$. \checkmark
\qedhere
\end{proof}
\end{theorem}
